AnalistaProgramador.java 
import java.util.List;
import java.util.ArrayList;

public class AnalistaProgramador {

    private List<String> skills = new ArrayList<>();

    public static void main(String[] args) {
        AnalistaProgramador dev = new AnalistaProgramador();

        String[] tecnologias = {
            "Java", "Python", "Spring",
            "Angular", "Docker", "AWS"
        };

        for (String tech : tecnologias) {
            dev.aprender(tech);
        }
    }

    public void aprender(String skill) {
        this.skills.add(skill);
    }
}
